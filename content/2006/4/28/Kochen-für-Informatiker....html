Nahrung Reistopf(INT typ)
::Typ=Konstante für die Art des Reistopfes
:: 0 = normal
:: 1 = Asiatisch
:: 2 = scharf, mit Kidney-Bohnen

Const
Zutat
immer_nötig[]={'Salz','Pfeffer','Loorbeerblätter','Reis','Wasser','Speiseöl'};
//Ich verwende auf Grund der einfachen Portionierung gerne Reis im
Kochbeutel
//1 Beutel =125g, für die Varianten 0,1 jeweils 2 Beutel, für V2 nur
einen
Zubehör benötigt[]={'scharfes Messer','Topf','Pfanne','Holzlöffel'};
VAR
Zutat Gemüse[];
Zutat Gewürze[];
Zutat Eiweisquelle;
{
if Typ=0 then
{
Gemüse={'Paprika','Möhren','Erbsen'};//naja, zur Not reicht auch ne
Dose Erbsen und Möhren
//(3 PortionenAbtropfgewicht
ca. 250g)
Gewürze={'Knoblachgranulat','Chillipulver','Kräuter der Provance'};
Eiweisquelle={'Hähnchenbrustfilet'}; //idealerweise 1 Packung zu
400-450g für ca. 3 Portionen
}
if Typ=1 then
{
Gemüse={'Asiatisches Pfannengemüse (Dose)'} //idealer wären
natürlich frische Sprossen und Pilze
//(1 Dose=3
Portionen,Abtropfgewicht ca. 250g)
Gewürze={'Chillipulver','Sojasoße','Basilikum'};
Eiweisquelle={'Hähnchenbrustfilet'}; //idealerweise 1 Packung zu
400-450g für ca. 3 Portionen
}
if Typ=2 then
{
Gemüse={'Dose Chilli Bohnen'}; //Kidney Bohnen in mexikanischer,
scharfer Soße
//(1 Dose=2 Portionen,Füllmenge ca. 400g)
Gewürze={'Chillipulver','diverse Kräuter'};
Eiweisquelle={'Dosenfisch'}; //ich verwendete 200g Heringsfilet in
Tomatensoße für 2 Portionen
//dies ist aber nicht wirklich ideal
}
wasser_aufsetzen('topf','wasser nach angaben auf Reispackung','pro 125g
Reis jeweils 1 Loorbeerblatt','kleine menge Salz');
while !wasser_kocht() do wait();
Reis_kochen(topf);
warten_und_vorbereiten(angabe_auf_reispackung - 5min,
kleinschndeiden(Eiweisquelle,scharfes messer));
if Typ=0|1 then
{
anbraten(Eiweisquelle,Pfanne,Speiseöl,Holzlöffel); //Holzlöffel
nur zum wenden verwenden... nicht mitbraten!^^
würzen(Eiweisquelle,Gewürze,'Salz','Pfeffer');
wait_until(Angabe auf reispackung-1min &&
Fleisch_komplett_durchgebraten)
anbraten(Gemüse); //einfach zusätzlich in die Pfanne geben
wait(1min);
anbraten(Reis); //Kochbeutel aus dem Wasser nehmen, aufschneiden
und in die Pfanne geben
//Loorbeerblätter und Wasser können entsorg
werden
nachwürzen(Gewürze);
}
else //Typ=2
{
anbraten(Gemüse,Eiweisquelle,Pfanne,Speiseöl,Holzlöffel);
//Holzlöffel nur zum wenden verwenden... nicht mitbraten!^^
würzen(Gewürze,'Salz','Pfeffer');
wait_until(Angabe auf reispackung)
anbraten(Reis); //Kochbeutel aus dem Wasser nehmen, aufschneiden
und in die Pfanne geben
//Loorbeerblätter und Wasser können
entsorg werden
nachwürzen(Gewürze);
}
return 'Pfanne mit Essen';
}
